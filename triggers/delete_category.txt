CREATE TRIGGER delete_category
ON Categories
INSTEAD OF DELETE
AS
BEGIN
	IF(NOT EXISTS (SELECT * FROM Categories_and_books WHERE deleted.cid = cid))
               DELETE 
        IF(deleted.host != 0)
        BEGIN
            @host = SELECT host FROM deleted
            UPDATE Categories_and_books SET host = @host WHERE host = (SELECT cid FROM deleted)
        END
END
